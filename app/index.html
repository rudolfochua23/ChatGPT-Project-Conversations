Setting Up Your Account

1

Open the App

Visit the AI Chat Consolidator URL (e.g. https://aichat.powerbyte.app). You'll see this landing page.

2

Create an Account

Click "Get Started" or "Create account". Fill in your email, choose a username (3-30 chars), and a password (min 12 chars).

3

Verify Your Email

Check your inbox for a 6-digit verification code. Enter it on the verification screen. The code expires in 15 minutes.

After verification you're automatically logged in. Browse conversations in the left panel, read messages in the center, and view code snippets on the right.

5

Forgot Your Password?

Click "Forgot password?" on the sign-in screen. Enter your email and check your inbox for a reset link. The link expires in 1 hour.

Setting Up Tampermonkey

1

Install Tampermonkey Extension

Install Tampermonkey for your browser: tampermonkey.net. Available for Chrome, Firefox, Edge, and Safari.

2

Create a GitHub Personal Access Token

Go to GitHub → Settings → Developer settings → Personal access tokens → Tokens (classic). Create a new token with the repo scope. Copy the token — you'll need it in the script.

3

Create a GitHub Repository

Create a repository (e.g. AI-Chats) where conversations will be stored. It can be private or public.

4

Add the Script

Open the Tampermonkey dashboard → click "Create a new script" → delete the default template → paste the script below → update the CONFIG section with your GitHub token, username, and repo name → save (Ctrl+S).

5

Start Chatting

Visit chatgpt.com or claude.ai. You'll see a "Save Chat" button in the bottom-left corner. Conversations auto-save every 3 minutes when changes are detected.

Sample Tampermonkey Script

// ==UserScript==
// @name         AI Chat Archiver
// @namespace    https://aichat.powerbyte.app
// @version      1.0
// @description  Auto-save ChatGPT & Claude AI conversations to GitHub
// @match        https://chatgpt.com/*
// @match        https://chat.openai.com/*
// @match        https://claude.ai/*
// @grant        GM_xmlhttpRequest
// @grant        GM_setValue
// @grant        GM_getValue
// @connect      api.github.com
// ==/UserScript==

(function () {
  'use strict';

  // ━━━ CONFIGURATION (edit these) ━━━━━━━━━━━━━━━━━━
  const CONFIG = {
    GITHUB_TOKEN: 'ghp_xxxxxxxxxxxxxxxxxxxx', // Your GitHub PAT
    GITHUB_OWNER: 'your-username',             // GitHub username
    GITHUB_REPO:  'AI-Chats',                  // Repository name
    BRANCH:       'main',
    LOG_PATH:     'logs/conversations',
    AUTOSAVE_MS:  3 * 60 * 1000,               // Auto-save every 3 minutes
  };

  // ━━━ PLATFORM DETECTION ━━━━━━━━━━━━━━━━━━━━━━━━━━
  function detectPlatform() {
    const h = location.hostname;
    if (h.includes('chatgpt') || h.includes('openai')) return 'chatgpt';
    if (h.includes('claude')) return 'claudeai';
    return null;
  }

  // ━━━ EXTRACTION ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  function extractChatGPT() {
    const msgs = [];
    document.querySelectorAll('[data-message-author-role]').forEach(el => {
      const role = el.getAttribute('data-message-author-role');
      const prose = el.querySelector('.markdown, .prose, [class*="markdown"]');
      const text = (prose?.innerText || el.innerText || '').trim();
      if (text) msgs.push({ role: role === 'user' ? 'User' : 'Assistant', text });
    });
    return msgs;
  }

  function extractClaude() {
    const msgs = [];
    document.querySelectorAll('[data-testid="user-message"]').forEach(el => {
      const text = (el.innerText || '').trim();
      if (text) msgs.push({ role: 'User', text });
    });
    document.querySelectorAll('.font-claude-message').forEach(el => {
      const clone = el.cloneNode(true);
      clone.querySelectorAll('button, [role="button"], svg, form, ' +
        '[class*="action"], [class*="toolbar"], [class*="copy"]')
        .forEach(n => n.remove());
      const text = (clone.innerText || '').trim();
      if (text && text.length > 10) msgs.push({ role: 'Assistant', text });
    });
    return msgs;
  }

  // ━━━ MARKDOWN FORMAT ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  function getTitle() {
    return document.title.replace(/ \|.*$/, '').trim() || 'Untitled';
  }

  function getConvId() {
    const m = location.pathname.match(/\/([a-zA-Z0-9-]+)$/);
    return m ? m[1] : 'conv_' + Date.now();
  }

  function toMarkdown(platform, messages) {
    const title = getTitle(), id = getConvId(), now = new Date().toISOString();
    let md = '# ' + (platform === 'chatgpt' ? 'ChatGPT' : 'Claude AI') +
             ' Conversation Log\n';
    md += '- Conversation ID: ' + id + '\n';
    md += '- Title: ' + title + '\n- Captured: ' + now + '\n';
    md += '- Platform: ' + platform + '\n- URL: ' + location.href + '\n---\n\n';
    messages.forEach(m => { md += '## ' + m.role + '\n\n' + m.text + '\n\n'; });
    return { md, title, id, platform };
  }

  // ━━━ GITHUB API ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  function slug(s) {
    return s.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/^-|-$/g, '').slice(0, 60);
  }

  function gmFetch(url, method, body) {
    return new Promise((resolve, reject) => {
      GM_xmlhttpRequest({
        method, url,
        headers: {
          Authorization: 'token ' + CONFIG.GITHUB_TOKEN,
          'Content-Type': 'application/json',
          Accept: 'application/vnd.github.v3+json',
        },
        data: body ? JSON.stringify(body) : undefined,
        onload: r => r.status < 300
          ? resolve(JSON.parse(r.responseText || '{}'))
          : reject(new Error('GitHub ' + r.status)),
        onerror: reject,
      });
    });
  }

  async function pushToGitHub(content, filePath, msg) {
    const url = 'https://api.github.com/repos/' + CONFIG.GITHUB_OWNER +
                '/' + CONFIG.GITHUB_REPO + '/contents/' + filePath;
    let sha = null;
    try { sha = (await gmFetch(url, 'GET')).sha; } catch {}
    const body = {
      message: msg,
      content: btoa(unescape(encodeURIComponent(content))),
      branch: CONFIG.BRANCH,
    };
    if (sha) body.sha = sha;
    return gmFetch(url, 'PUT', body);
  }

  // ━━━ SAVE ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  function simpleHash(s) {
    let h = 0;
    for (let i = 0; i < s.length; i++) h = ((h << 5) - h + s.charCodeAt(i)) | 0;
    return h;
  }

  async function save() {
    const platform = detectPlatform();
    const messages = platform === 'chatgpt' ? extractChatGPT() : extractClaude();
    if (!messages.length) { toast('No messages found.'); return; }
    const { md, title, id } = toMarkdown(platform, messages);
    const path = CONFIG.LOG_PATH + '/' + platform + '/' + slug(title) + '__' + id + '.md';
    toast('Saving...');
    try {
      await pushToGitHub(md, path, 'Auto-save conversation: ' + title);
      GM_setValue('hash_' + id, simpleHash(md));
      toast('Saved! (' + messages.length + ' messages)');
    } catch (e) { toast('Save failed: ' + e.message); }
  }

  // ━━━ UI ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  function toast(msg) {
    let el = document.getElementById('ai-archiver-toast');
    if (!el) {
      el = document.createElement('div');
      el.id = 'ai-archiver-toast';
      Object.assign(el.style, {
        position:'fixed',bottom:'20px',right:'20px',zIndex:'99999',
        background:'#1a2130',color:'#e5ecff',border:'1px solid #7ba2ff',
        borderRadius:'8px',padding:'10px 16px',fontSize:'14px',
        fontFamily:'system-ui',maxWidth:'320px',boxShadow:'0 4px 20px rgba(0,0,0,.4)',
      });
      document.body.appendChild(el);
    }
    el.textContent = msg; el.style.display = 'block';
    clearTimeout(el._t);
    el._t = setTimeout(() => { el.style.display = 'none'; }, 3000);
  }

  // ━━━ INIT ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  if (!detectPlatform()) return;
  setTimeout(() => {
    const btn = document.createElement('button');
    btn.textContent = 'Save Chat';
    Object.assign(btn.style, {
      position:'fixed',bottom:'20px',left:'20px',zIndex:'99999',
      background:'#7ba2ff',color:'#0b1020',border:'none',borderRadius:'8px',
      padding:'8px 16px',fontSize:'14px',fontWeight:'700',cursor:'pointer',
    });
    btn.onclick = save;
    document.body.appendChild(btn);
    toast('AI Chat Archiver active');
  }, 2000);

  // Auto-save when content changes
  setInterval(() => {
    const p = detectPlatform();
    const msgs = p === 'chatgpt' ? extractChatGPT() : extractClaude();
    if (!msgs.length) return;
    const { md, id } = toMarkdown(p, msgs);
    if (simpleHash(md) !== GM_getValue('hash_' + id, 0)) save();
  }, CONFIG.AUTOSAVE_MS);
})();
